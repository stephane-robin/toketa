Lorsque le joueur doit deviner

L'objectif est de créer un programme Python qui demande à l'utilisateur de deviner un pays en Europe, suite à 3 indications données par le programme. Le joueur dispose donc de 3 tentatives pour trouver le pays et sa réponse est acceptée en minuscules ou avec la première lettre en majuscule.

pays = [

import random

trouve = False 
nbre_tentatives = 0

pays = [
    {
        'nom': 'Allemagne',
        'langue': 'l\'allemand',
        'personnalite': 'Goethe',
        'capitale': 'Bonn'
    },
    {
        'nom': 'Autriche',
        'langue': 'l\'allemand',
        'personnalite': 'Freud',
        'capitale': 'Vienne'
    },
    {
        'nom': 'Suisse',
        'langue': 'l\'allemand',
        'personnalite': 'Rousseau',
        'capitale': 'Bern'
    }
]

questions = [
    "La langue parlée dans ce pays est : ",
    "Une personnalité célèbre de ce pays est : ",
    "La capitale de ce pays est : "
]

cles = [
    'langue',
    'personnalite',
    'capitale'
]

paysATrouver = random.choices(pays)[0]

i = 0
while trouve == False and nbre_tentatives < 3:
    reponse = input(questions[i] + paysATrouver[cles[i]] + ". Quel est ce pays ? ").capitalize()
    nbre_tentatives += 1
    if reponse == paysATrouver['nom']:
        trouve = True
    i += 1

if trouve == True: 
    print("Bravo, vous avez trouvé en " + str(nbre_tentatives) + " tentative(s).")
else:
    print("Dommage, vous avez perdu. La réponse était : " + paysATrouver['nom'])
